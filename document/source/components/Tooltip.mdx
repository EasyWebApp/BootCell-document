```TSX
import { render, createCell } from 'web-cell';
import { TooltipBox } from 'boot-cell/source/Prompt/Tooltip';
import { Button } from 'boot-cell/source/Form/Button';

render(['top', 'right', 'bottom', 'left'].map(position => (
    <TooltipBox position={position} text="Tooltip on top">
        <Button kind="secondary">Tooltip on {position}</Button>
    </TooltipBox>
)));
```

And with custom HTML added:

```TSX
import { render, createCell, Fragment } from 'web-cell';
import { TooltipBox } from 'boot-cell/source/Prompt/Tooltip';
import { Button } from 'boot-cell/source/Form/Button';

render(
    <TooltipBox text={
        <Fragment>
            <em>Tooltip</em>{' '}
            <u>with</u>{' '}
            <b>HTML</b>
        </Fragment>
    }>
        <Button kind="secondary">Tooltip on left</Button>
    </TooltipBox>
);
```
